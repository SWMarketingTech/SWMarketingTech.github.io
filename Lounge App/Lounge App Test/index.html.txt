  <script>
    let currentSlide = 0;
    let isDragging = false;
    let startX = 0;
    let translateX = 0;
    const totalSlides = 4;

    const carousel = document.getElementById('carousel');
    const carouselTrack = document.getElementById('carouselTrack');

    function updateCarousel() {
      carouselTrack.style.transform = `translateX(-${currentSlide * 100}%)`;
      
      // Update indicators
      for (let i = 0; i < totalSlides; i++) {
        const indicator = document.getElementById(`indicator-${i}`);
        if (i === currentSlide) {
          indicator.classList.remove('bg-white/30');
          indicator.classList.add('bg-white');
        } else {
          indicator.classList.remove('bg-white');
          indicator.classList.add('bg-white/30');
        }
      }
      
      // Animate bars when on slide 4 (index 3)
      if (currentSlide === 3) {
        setTimeout(() => {
          for (let i = 1; i <= 5; i++) {
            const bar = document.getElementById(`bar-${i}`);
            const height = bar.getAttribute('data-height');
            bar.style.height = height + '%';
          }
        }, 300);
      } else {
        // Reset bars when leaving slide 4
        for (let i = 1; i <= 5; i++) {
          const bar = document.getElementById(`bar-${i}`);
          bar.style.height = '0%';
        }
      }
    }

    function handleStart(e) {
      isDragging = true;
      startX = e.pageX || e.touches[0].pageX;
      carouselTrack.style.transition = 'none';
    }

    function handleMove(e) {
      if (!isDragging) return;
      const currentX = e.pageX || e.touches[0].pageX;
      translateX = currentX - startX;
      carouselTrack.style.transform = `translateX(calc(-${currentSlide * 100}% + ${translateX}px))`;
    }

    function handleEnd() {
      if (!isDragging) return;
      isDragging = false;
      carouselTrack.style.transition = 'transform 0.3s ease-out';
      
      if (Math.abs(translateX) > 50) {
        if (translateX > 0 && currentSlide > 0) {
          currentSlide--;
        } else if (translateX < 0 && currentSlide < totalSlides - 1) {
          currentSlide++;
        }
      }
      
      translateX = 0;
      updateCarousel();
    }

    // Mouse events
    carousel.addEventListener('mousedown', handleStart);
    carousel.addEventListener('mousemove', handleMove);
    carousel.addEventListener('mouseup', handleEnd);
    carousel.addEventListener('mouseleave', handleEnd);

    // Touch events
    carousel.addEventListener('touchstart', handleStart);
    carousel.addEventListener('touchmove', handleMove);
    carousel.addEventListener('touchend', handleEnd);

    // Initialize
    updateCarousel();
  </script>
